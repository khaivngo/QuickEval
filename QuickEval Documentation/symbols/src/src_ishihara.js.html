  2 /**
  3  * Listens for user clicking submit, if clicked input values are checked,
  4  * if they matches the ajax will be run updating whether user have problem
  5  * with perceiving colours.
  6  */
  7 $(document).ready(function() {
  8     $('#submit').click(function() {
  9         if (checkValues()) {
 10             $.ajax
 11                     ({
 12                         url: 'ajax/observer/updateColourBlindness.php',
 13                         async: false,
 14                         success: function(data) {
 15                         },
 16                         error: function(request, status, error) {
 17                             alert(request.responseText);
 18                         }
 19                     });
 20         }
 21     });
 22     setUpModeMenu('.observer-mode');
 23 
 24 });
 25 /**
 26  * Check if users input is the correct values
 27  * @returns {Boolean}
 28  */
 29 function checkValues() {        //Input values matches images
 30     if ($('#picture1').val() == 12 && $('#picture2').val() == 5 && $('#picture3').val() == 8 && $('#picture4').val() == 26) {
 31         $('#submit-div').on('click', function() {
 32             $.Notify({             //notifies user
 33                 content: "<br><br>        You appear to perceive colours normally",
 34                 style: {background: '#32CD32'},
 35                 width: '200px',
 36                 height: '120px',
 37                 timeout: '6000'
 38             });
 39         });
 40         setTimeout(function() {  //sends user back to main page after a short delay
 41             window.location = 'index.php';
 42         }, 3000);
 43 
 44         return true;
 45     }
 46     else {
 47 
 48         $('#submit-div').on('click', function() {
 49             $.Notify({          //notifies user
 50                 content: "<br><br>        Your might have problems with perceiving colours properly",
 51                 style: {background: '#A9A9A9'},
 52                 width: '200px',
 53                 height: '120px',
 54                 timeout: '6000'
 55             });
 56         });
 57 
 58         setTimeout(function() {     //sends user back to main page after a short delay
 59             window.location = 'index.php';
 60         }, 3000);
 61         return false;
 62     }
 63 }
