447 /**
448  * Sets up listener for level-buttons on stepper. Skips first click using the loaded variable, as the stepper
449  * clicks once on last step as its loaded
450  * @returns {undefined}
451  */
452 function setUpStepperListener() {
453     $(document.body).off('click', 'li'); //Makes sure only one listener for stepper is active at a time
454     $(document.body).on('click', 'li', function() { //Sets up listener for stepper to enable navigating by clicking
455         if (loaded == 1) { // steps
456             var diff = ($(this).index() - $('.current').index()); //Gets amount of steps from current to selected
457             if (diff == 1) { //Adds step if differance was 1
458                 $('#stepper').stepper('next'); //Only 1 step forward is allowed at a time
459             } else if (diff < 0) { //Removes steps if differance was negative
460                 for (var i = 0; i > diff; i--) {
461                     $('#stepper').stepper('prior');
462                 }
463             }
464         }
465         loaded = 1; //Updates loaded
466     });
467 }

469 /**
470  * Swaps the position of two elements
471  * @param {type} a The first element
472  * @param {type} b The second element
473  * @returns {undefined}
474  */
475 function swapElements(a, b) {
476     aP = a.parent(); //Gets the elements parent
477     bP = b.parent();
478     aP.append(b); //Appends the new elements
479     bP.append(a);
480     a.css('top', 0); //Prevents weird positioning after dropping
481     a.css('left', 0);
482 }

484 /**
485  * Enables dragging and dropping on elements with classes draggable and droppable.
486  * Dropping one element on another switches their position via the swapElements function
487  * @param {type} bool Set to true if managing image-queues and not experimentqueue
488  * @returns {undefined}
489  */
490 function setUpDragNDrop(bool) {
491     //Adds dragging
492     $('.draggable').draggable({
493         revert: "invalid", //Return element to its position if drop on non-dropzone
494         stack: "div", //Dragged elements are always on top
495         containment: '.steps',
496         //Used on picture queue. Adds droppable only when dragging is started, to contain droppable zones only to
497         // elements within the same picture set
498         start: function(event, ui) {
499             if (bool == true) {
500                 $(this).parent().siblings().children('.input-control').droppable({
501                     drop: function(event, ui) {
502                         swapElements(ui.draggable, $(this)); //Swaps the elements
503                         $('.ex-pair.ui-droppable').droppable('destroy');
504                     }
505                 });
506             }
507         },
508         //Used on picture queue. Removes droppable functionality when dragging stops, to prevent enabling dragging between
509         // picture-sets
510         stop: function(event, ui) {
511             if (bool == true) {
512                 $('.ex-pair.ui-droppable').droppable('destroy');
513             }
514         }
515     });
516     //Sets all elements with class droppable. Used on experimentqueue only
517     if (bool != true) {
518         //Adds dropping
519         $('.droppable').droppable({
520             drop: function(event, ui) {
521                 if (checkDragNDropElements($(this), ui.draggable)) {
522                     swapQueue(ui.draggable, $(this));
523                 } else {
524                     checkDragNDropOtherElements($(this), ui.draggable); //Checks if user swaps a imageset with nonimageset
525                 }
526                 swapElements(ui.draggable, $(this)); //Swaps the elements
527             }
528         });
529     }
530 }

586 /**
587  * Swaps the elements on dom level
588  * @author Paolo Bergantino http://stackoverflow.com/questions/698301/is-there-a-native-jquery-function-to-switch-elements
589  * @param {type} to
590  * @returns {jQuery.fn@call;each}
591  */
592 jQuery.fn.swapWith = function($to) {
593     return this.each(function() {
594         var copy_to = $($to).clone(true);
595         var copy_from = $(this).clone(true);
596         $($to).replaceWith(copy_from);
597         $(this).replaceWith(copy_to);
598     });
599 };

682  * @param {type} $b second element

685 function checkDragNDropOtherElements($a, $b) {

696 /**
697  * Fixes queue when imageset and non imageset is switched
698  * @param {type} $a active imageset
699  * @param {type} $b non imageset, either instruction or inactive imageset
700  * @returns {undefined}
701  */
702 function swapImageSetAndElement($a, $b) {
703     var activeImageSet = $a.closest('.image-set-div').parent(); //The imageset
704     var index = getImageSetIndex($a);
705     var otherElement = $b.closest('.image-set-div, .instruction-field, .instruction-field-history').parent(); //Other element
706     var elementsBetween; //Elements between imageset and other element
707     var direction; //True if up false if down
708     var activeQueue = $('.ex-image-pair').eq(index); //Queue for active imageset
709     //Checks which direction queue has to be moved and saves elements between dragged and dropped element
710     if (activeImageSet.index() < otherElement.index()) { //Active imageset is above other element
711         elementsBetween = activeImageSet.nextUntil(otherElement); //Elements between
712         direction = false; //Queue needs to be moved down
713     } else { //Active imageset is below other element
714         elementsBetween = activeImageSet.prevUntil(otherElement); //Elements between
715         direction = true; //Queue needs to be moved up
716     }
717     var imageSets = $('select > option:selected:not(:disabled)').closest('.image-set-div').parent();
718     var notActiveImageSets = elementsBetween.not(imageSets); //Gets elements who aren't active imagesets
719     var imageSetsBetween = elementsBetween.not(notActiveImageSets); //Removes other elements to get only active imagesets between
720     //Moves the queue to the right position after imageset has been moved
721     if (direction == false) { //Imagequeue has to be moved down
722         $('.ex-image-pair').eq(index + imageSetsBetween.length).after(activeQueue);
723     } else { //Imagequeue has to be moved up
724         $('.ex-image-pair').eq(index - imageSetsBetween.length).before(activeQueue);
725     }
726 }

1072 /**
1073  * Makes sure all pairs are matched with each other
1074  * @return {int} 0 if no pairs missing, else a positive value
1075  */
1076 function checkMatchedPairs() {
1077     var missingPairs = 0;
1078     $('.image-pairs').each(function() {
1079         var images = new Array();
1080         var firstSet = new Array(); //images left in pair
1081         var secondSet = new Array(); //images right in pair
1082         var check = 0; //Local check if images are missing in current imageset
1083         var instances;
1084         //Get all images
1085         $(this).find('.first-image').eq(0).children(':not(:disabled)').each(function() {
1086             images.push($(this).attr('imageid'));
1087         });
1088         //Gets all first images
1089         $(this).find('.first-image').each(function() {
1090             firstSet.push($(this).children('option:selected:not(:disabled)').attr('imageid'));
1091         });
1092         //Gets all second images
1093         $(this).find('.second-image').each(function() {
1094             secondSet.push($(this).children('option:selected:not(:disabled)').attr('imageid'));
1095         });
1096         //Removes duplicates/the same images matched with eachother
1097         for (var i = 0; i < firstSet.length; i++) {
1098             if (firstSet[i] == secondSet[i]) {
1099                 firstSet.splice(i, 1);
1100                 secondSet.splice(i, 1);
1101                 i--;
1102             }
1103         }
1104         //Get all instances 
1105         instances = countImages(firstSet.concat(secondSet));
1106         //Checks if there are missing instances
1107         for (var i = 0; i < images.length; i++) {
1108             if ($.inArray(images[i], instances[0]) == -1) {
1109                 check++;
1110                 missingPairs++;
1111             }
1112         }
1113         //Checks if the count for each instanced image is amount-1
1114         for (var i = 0; i < instances[1].length; i++) {
1115             if (instances[1][i] < images.length - 1) {
1116                 check++;
1117                 missingPairs++;
1118             }
1119         }
1120         if (check != 0) {
1121             $(this).append('<div><div class="span3" style="margin: 20px 20px"></div>' + '<div class="bg-red notice image-queue-check marker-on-top span1">' + 'Not all images are matched! Press finish again to continue, or this box to check again' + '</div></div>');
1122         }
1123     });
1124     return missingPairs;
1125 }
1126 /**
1127  *  Counts and returns amount of images in array with objects
1128  *  @author Sime Vidas http://stackoverflow.com/questions/5667888/counting-occurences-of-javascript-array-elements
1129  *
1130  **/
1131 function countImages($arr) {
1132     var a = [],
1133         b = [],
1134         prev;
1135     $arr.sort();
1136     for (var i = 0; i < $arr.length; i++) {
1137         if ($arr[i] !== prev) {
1138             a.push($arr[i]);
1139             b.push(1);
1140         } else {
1141             b[b.length - 1]++;
1142         }
1143         prev = $arr[i];
1144     }
1145     return [a, b];
1146 }
