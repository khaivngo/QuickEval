162 /**
163  * Loops through all images in sortable and getting their id and position
164  * @returns {undefined}
165  */
166 function updateSortablePosition() {
167     var ids = $('#rating-images > div').map(function(i) {       //getting all elements id.
168         return this.id;
169     }).get();
170 
171     var pos = $('#rating-images > div').map(function(i) {       //getting all the elements position.
172         return jQuery(this).index();
173     }).get();
174 
175     var arrayLength = ids.length;              //get's length of the id array.
176 
177     for (var i = 0; i < arrayLength; i++) {             //goes through array and calls function which sets 
178         // new position indicator for each.
179         setPosition(ids[i], pos[i] + 1); //pos+1 because it starts on 0.
180     }
181 }

216 /**
217  * Receives objects which contains all images and their data
218  * @returns {undefined}
219  */
220 function loadReproductionsSortable(data) {
221     var length;
222     $('#rating-images').empty();                //empties div for the next pictures to be loaded
223     length = Object.keys(data).length - 1;
224 
225     for (var i = 1; i <= length; i++) {                         //goes through all objects getting their data.
226         var reproductionImageUrl = data[i].url;                 //getting url
227         var reproductionPictureOrder = data[i].pictureOrderId;  //getting id
228 
229         if (i == 1) {              //first picture gets loaded into left panner.
230             loadImageIntoPanner(reproductionPictureOrder, reproductionImageUrl, "left");
231         }
232 
233         if (i == 2) {              //second picture get loaded into right panner.
234             loadImageIntoPanner(reproductionPictureOrder, reproductionImageUrl, "right");
235         }
236 
237         //each picture is appended to the sortable
238         $('#rating-images').append('<div class="image-position" id=' + i + '><a href="#" id="panner-side' + i + '" target="" class="panner-side-left"></a><a href="#" id="panner-side' + i + '" target="" class="panner-side-right"></a><p class="style-p" >1</p><img src=' + reproductionImageUrl + ' id=' + reproductionPictureOrder + ' ><br><span id="initial-position">' + i + '</span></div>');
239     }
240     $('a').nextAll('.panner-side-left').hide();
241     $('a').prevAll('.panner-side-left').hide();
242 
243     $('a').nextAll('.panner-side-right').hide();
244     $('a').prevAll('.panner-side-right').hide();
245 
246     //indicates which two pictures are default loaded into panner. (Always number 1 and 2)
247     pictureInPanner("panner-side1", null, "left");
248     pictureInPanner("panner-side2", null, "right");
249 
250     updateSortablePosition();
251 }

284 /**
285  * Goes to next step in current experiment
286  * @returns {undefined}
287  */
288 function loadExperiment() {
289     var data = getNextInExperimentForObserver();
290 
291     if (data['type'] == "experimentinstruction") {              //is instruction
292         var instruction = data['experimentinstruction'];        //get's the instruction
293         loadInstruction(instruction);                           //sends instruction to function for loading
294         onlyInstruction();                                      //displays instruction to user
295         loadExperiment();                                       //goes to the next step which in most cases is pictures.
296     }
297 
298     if (data['type'] == "pictureQueue") {                       //is picture set 
299 
300         if (ratingRunned == 1) {
301             postRating();
302         }
303 
304         loadReproductionsSortable(data);                        //get's all images in the experiment
305         var originalImageUrl = data[1]['originalUrl'].url;      //getting url of original image
306         loadOriginal(originalImageUrl);                         //loading original picture 
307 
308         ratingRunned = 1;
309 
310     }
311 
312     if (data['type'] == "finished") {                           //user is finished
313         experimentComplete(experimentId);
314         finished();
315         $("#button-finished").text("Finish");
316         $('#contactArea').empty();                              //changes content of popup
317         $('#continue').hide();
318         $('#contactArea').append("You have finished, thank you for your time <br><br> Click Quit to return to front page.");
319         // $('#cancel-experiment').trigger('click');
320     }
321 }

328 /**
329  * Applies a indicator for which picture in is loaded into panner
330  * @param {type} pictureId id of picture
331  * @param {type} side which panner, left or right
332  * @returns {undefined}
333  */
334 function pictureInPanner(aId, draggableId, side) {
335     var imagePannerIdLeft = $('#pan1').find('img').attr('pictureOrderId');
336     var imagePannerIdRight = $('#pan2').find('img').attr('pictureOrderId');
337     var initialPositionCheck = $('image-position').find()
338 
339     if (side == "left") {
340         //  $('#' + aId + '').show();
341 
342         $('a').nextAll('.panner-side-left').hide();
343         $('a').prevAll('.panner-side-left').hide();
344 
345         if (imagePannerIdLeft != draggableId) {
346             $('#' + aId + '').show();
347         }
348 
349         if (imagePannerIdLeft == draggableId) {
350             $('#' + aId + '').show();
351         }
352 
353         if (lastRight == lastLeft) {
354             $('#' + lastRight + '').removeClass("panner-side-right");
355         }
356 
357         if (imagePannerIdLeft == draggableId && imagePannerIdRight == draggableId) {
358             $('#' + aId + '').addClass("panner-side-left");
359             $('#' + aId + '').next().addClass("panner-side-right");
360             $('#' + aId + '').show();
361             $('#' + aId + '').next().show();
362         }
363 
364         if (draggableId == "null") {
365             $('#' + aId + '').next().show();
366         }
367 
368         lastLeftId = draggableId;
369         lastLeft = aId;
370         $("#rating-images").sortable("refresh");
371     }
372     else if (side == "right") {
373 
374         $('a').nextAll('.panner-side-right').hide();
375         $('a').prevAll('.panner-side-right').hide();
376 
377         if (imagePannerIdRight != draggableId && draggableId != null) {
378             $('#' + aId + '').addClass("panner-side-right");
379             $('#' + aId + '').show();
380         }
381 
382         if (lastRight == lastLeft && draggableId != null) {
383             $('#' + lastLeft + '').removeClass("panner-side-left");
384         }
385 
386         if (imagePannerIdLeft == draggableId && imagePannerIdRight == draggableId) {
387             $('#' + aId + '').addClass("panner-side-left");
388             $('#' + aId + '').next().addClass("panner-side-right");
389             $('#' + aId + '').show();
390             $('#' + aId + '').next().show();
391 
392             $('a').nextAll('.panner-side-right').hide();
393         }
394 
395         if (imagePannerIdLeft != lastRightId) {          //CHANGE TO == when handle for ID.
396             $('#' + lastRight + '').hide();
397             $('#' + aId + '').next().show();
398         }
399 
400         if (draggableId == null) {
401             $('#' + aId + '').next().show();
402         }
403 
404         lastRighId = draggableId;
405         lastRight = aId;
406         $("#rating-images").sortable("refresh");
407     }

422 /**
423  * Loops through all matching divs and checks,
424  * whether they have been visited or not.
425  * Returns true if all have been visited(touched).
426  * @returns {Boolean} Returns false if one is not visited.
427  */
428 function isAllVisited() {
429     var numItems = $('.image-position').length;     //gets the number of divs with particular class
430 
431     for (var i = 1; i <= numItems; i++) {
432         var hasClass = $('image-position, #' + i + '').hasClass('touched');
433         if (!hasClass) {
434             return false;           //one div has not been touched.
435         }
436     }
437     return true;                //all div have been visited.
438 }
