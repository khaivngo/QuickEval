164 /**
165  * Populates experiment list based on institute/organization/scientist and experiment search field.
166  * Gets random experiments if no experiment is searched for
167  * @param {type} $experiment Experiment to be searched for
168  * @returns {undefined}
169  */
170 function getExperiments($experiment) {
171 
172     var institute = $('#select-institution a.active:not(.all)').text();
173     var organization = $('#select-organization a.active:not(.all)').text();
174     var scientist = $('#select-scientist a.active:not(.all)').text();
175     $.ajax({
176         url: 'ajax/observer/getExperiments.php',
177         data: {
178             'experiment': $experiment,
179             'organization': organization,
180             'institute': institute,
181             'scientist': scientist
182         },
183         type: 'post',
184         async: false,
185         dataType: 'json',
186         success: function(data) {
187             $('#select-experiment a').remove();
188             emptyExperimentData();
189             if (data !== 0) {
190                 for (var i = 0; i < data.length; i++) {
191                     $('#select-experiment').append('<a class="list experiment" eid="' + data[i]['id'] +
192                         '" method="' + data[i]['experimentType'] + '" href="#">' +

194                         '<span id="experiment' + i + '" class="list-title">' + data[i]['title'] + '</span>' +
195                         '</div>' +
196                         '</a>');
197                 }
198             }
199         },
200         complete: function() {
201             displayByMethod($('#select-method').children('.active'));
202             $('#select-experiment').listview().init();
203         },
204         error: function(request, status, error) {
205             alert(request.responseText);
206         }

345 /**
346  * Gets and displays experiment data and "start experiment" or "ishihara" button
347  * @param {type} $eId
348  * @param {type} $type only returns data when 1
349  * @returns {undefined}
350  */
351 function getExperimentData($eId, $type, $invite) {
352     var result = 0;
353     $.ajax({
354         url: 'ajax/observer/getExperimentData.php',
355         data: {
356             'experiment': $eId,
357             'invite': $invite
358         },
359         type: 'post',
360         async: false,
361         dataType: 'json',
362         success: function(data) {
363             if (data[0] !== 0) {
364                 if ($type == 1) {
365                     result = data[0];
366                 } else {
367                     emptyExperimentData();
368                     $('#experiment-title').append(data[0]['title']);
369                     $('#experiment-info').append(data[0]['shortDescription']);
370                     $('#experiment-text').append(data[0]['longDescription']);
371 
372                     if (data[1] !== 0) {
373                         setUpCustomFields(data[1]);
374                     }
375 
376                     if (data[0]['allowColourBlind'] == 0 && checkColourBlind() == false) {
377                         $('#experiment-buttons').append('<button id="bottom-button" class="button success colourblind" style="margin 20px 0; float:right" ' +
378                             'eid="' + data[0]['id'] + '">Start Colourblind Test</button>')
379                     } else {
380                         $('#experiment-buttons').append('<button id="bottom-buttom"' +
381                             'class="button success start-experiment" style="margin 20px 0; float:right"' +
382                             'eid="' + data[0]['id'] + '">Start Experiment</button>');
383                     }
384                 }
385             }
386 
387         },
388         error: function(request, status, error) {
389             alert(request.responseText);
390         }
391     });
392     return result;
393 }

466 /**
467  * Starts experiment with invite url
468  * @return {[type]} [description]
469  */
470 function startExperiment() {
471     saveFieldValues();
472     var url;
473     var data = getExperimentData($('.start-experiment').attr('eid'), 1);
474     var invite = "";
475     if ($('#invite').val() != undefined) {
476         invite = '?invite=' + $('#invite').val();
477     }
478     var type = data['experimentType'];
479 
480     switch (type) {
481         case '1':
482             url = 'rankorderexperiment.php';
483             break;
484 
485         case '2':
486             url = 'pairComparisonExperiment.php';
487             break;
488 
489         case '3':
490             url = 'categoryexperiment.php';
491             break;
492     }
493 
494     url+=invite;    //adds invite hash if set
495 
496     var form = $('<form action="' + url + '" method="post" target="_blank">' +
497         '<input type="hidden" name="experimentId" value="' + $('.start-experiment').attr('eid') + '" />' +
498         '</form>');
499     $('body').append(form);
500     $(form).submit();
501 }
