 53 /**
 54  * Inects a html-page into the right panel div.
 55  * $data is location of html file.
 56  */
 57 function inject($data) {
 58 
 59     $.ajax({
 60         async: false,
 61         url: $data,
 62         success: function(data) {
 63             $($('#right-panel')).html(data);
 64             $('#right-menu').empty();
 65         },
 66         error: function(request, status, error) {
 67             alert(request.responseText);
 68         }
 69     });
 70 }

 72 /**
 73  * Sets the $target button active in the left
 74  * sidebar.
 75  * $data is the jquery element to be set active.
 76  */
 77 function setActive($target) {
 78     $target.siblings().removeClass("active");
 79     $target.addClass("active");
 80 }
