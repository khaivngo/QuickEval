 67 /**
 68  * Get's the dimensions of the viewport.
 69  * @returns {viewport.Anonym$0}
 70  */
 71 function viewport() {
 72     var e = window;
 73     var a = 'inner';
 74     if (!('innerWidth' in window)) {
 75         a = 'client';
 76         e = document.documentElement || document.body;
 77     }
 78     return {width: e[ a + 'Width' ], height: e[ a + 'Height' ]};
 79 }

 81 /**
 82  * Formats time values and appends zero where it is needed.
 83  * @param {type} number the value
 84  * @param {type} length length of the value
 85  * @returns {pad.str|String}
 86  */
 87 function pad(number, length) {
 88     var str = '' + number;
 89     while (str.length < length) {
 90         str = '0' + str;
 91     }
 92     return str;
 93 }

 95 /**
 96  * Format times and returns value
 97  * @param {type} time in milliseconds
 98  * @returns {String} contains the whole time format
 99  */
100 function formatTime(time) {
101     var min = parseInt(time / 6000),
102             sec = parseInt(time / 100) - (min * 60),
103             hundredths = pad(time - (sec * 100) - (min * 6000), 2);
104     return (min > 0 ? pad(min, 2) : "00") + ":" + pad(sec, 2) + ":" + hundredths;
105 }

107 /**
108  * Returns operating system.
109  * @returns {String}
110  */
111 function getOs() {
112     var OSName = "unknown OS";
113     if (navigator.appVersion.indexOf("Win") != -1)
114         OSName = "Windows";
115     if (navigator.appVersion.indexOf("Mac") != -1)
116         OSName = "MacOS";
117     if (navigator.appVersion.indexOf("X11") != -1)
118         OSName = "UNIX";
119     if (navigator.appVersion.indexOf("Linux") != -1)
120         OSName = "Linux";
121 
122     return OSName;
123 }

126 /**
127  * http://stackoverflow.com/questions/10211145/getting-current-date-and-time-in-javascript
128  * @author Daniel Lee
129  * @returns {String} date and time
130  */
131 function getDateTime() {
132     var now = new Date();
133     var year = now.getFullYear();
134     var month = now.getMonth() + 1;
135     var day = now.getDate();
136     var hour = now.getHours();
137     var minute = now.getMinutes();
138     var second = now.getSeconds();
139     if (month.toString().length == 1) {
140         var month = '0' + month;
141     }
142     if (day.toString().length == 1) {
143         var day = '0' + day;
144     }
145     if (hour.toString().length == 1) {
146         var hour = '0' + hour;
147     }
148     if (minute.toString().length == 1) {
149         var minute = '0' + minute;
150     }
151     if (second.toString().length == 1) {
152         var second = '0' + second;
153     }
154     var dateTime = year + '-' + month + '-' + day + ' ' + hour + ':' + minute + ':' + second;
155     return dateTime;
156 }

179 /**
180  * Hides/show fullscreen buttons whether user are using IE.
181  * http://jsfiddle.net/9zxvE/383/
182  * @returns {undefined}
183  */
184 function IESpecific() {
185     var isOpera = !!window.opera || navigator.userAgent.indexOf(' OPR/') >= 0;
186     // Opera 8.0+ (UA detection to detect Blink/v8-powered Opera)
187     var isFirefox = typeof InstallTrigger !== 'undefined'; // Firefox 1.0+
188     var isSafari = Object.prototype.toString.call(window.HTMLElement).indexOf('Constructor') > 0;
189     // At least Safari 3+: "[object HTMLElementConstructor]"
190     var isChrome = !!window.chrome && !isOpera; // Chrome 1+
191     var isIE = /*@cc_on!@*/false || !!document.documentMode; // At least IE6
192 
193     if(isFirefox)   {
194         $('.image-position').switchClass("image-position", "image-position-moz", 0);
195         $('.panner-side-right').switchClass("panner-side-right", "panner-side-right-moz", 0);
196     }
197 
198     if (isIE == true) {
199 
200         $('#enter-fullscreen').addClass('iefix').remove();
201         $('#exit-fullscreen').addClass('iefix').remove();
202         $('#ie-info-fullscreen').show();
203 
204         $('#button-finished').removeClass('button-finished');
205         $('#button-finished').addClass('button-finished-IE');
206 
207         $('.image-position').each(function(i, obj) {
208             $(this).removeClass('image-position');
209             $(this).addClass('image-position-IE');
210         });
211 
212         $('#popupButtons2').addClass('popupButton-important-IE');
213         $('#rating').css("width","65%");
214         $('.panner-side-left').switchClass("panner-side-left", "panner-side-left-ie", 0);
215         $('.panner-side-right').switchClass("panner-side-right", "panner-side-right-ie", 0);
216 
217     }
218     else {
219         $('#ie-info-fullscreen').hide();
220     }
221 }

274 /**
275  * Enables fullscreens mode
276  * @returns {undefined}
277  */
278 function goFullscreen() {
279     var docElement, request;
280     docElement = document.documentElement;
281     request = docElement.requestFullScreen || docElement.webkitRequestFullScreen || docElement.mozRequestFullScreen || docElement.msRequestFullScreen;
282     if (typeof request != "undefined" && request) {
283         request.call(docElement);
284     }
285 }
286 
287 /**
288  * Exits fullscreen mode
289  * @returns {undefined}
290  */
291 function exitFullscreen() {
292     var docElement, request;
293     docElement = document;
294     request = docElement.cancelFullScreen || docElement.webkitCancelFullScreen || docElement.mozCancelFullScreen || docElement.msCancelFullScreen || docElement.exitFullscreen;
295     if (typeof request != "undefined" && request) {
296         request.call(docElement);
297     }
298 }
