1 /**
  2  * Starts a new experiment for a user, and this function
  3  * will make everything ready, and store the queue in session
  4  * @param  experimentId is the ID of the experiment to start.
  5  */
  6 function startNewExperimentForObserver(experimentId) {
  7     $.ajax
  8     ({
  9         url: 'ajax/observer/experimentQueue.php',
 10         async: false,
 11         data: {
 12         	'option':'startNewObserverExperiment',
 13         	'experimentId':experimentId
 14         },
 15         error: function(request, status, error) {
 16             alert(request.responseText);
 17         }
 18     });
 19 }

 21 /**
 22  * Gets the next position in the experiment for a observer.
 23  * @return instruction with text, the pictures which are to be shown,
 24  * or a text with "finished" that indicates that an experiment is over.
 25  * 
 26  */
 27 function getNextInExperimentForObserver() {
 28 	var returnData = new Array();
 29     $.ajax
 30     ({
 31         url: 'ajax/observer/experimentQueue.php',
 32         async: false,
 33         data: {
 34         	'option':'getNextPosition',
 35         },
 36         dataType: 'json',
 37         success: function(data)
 38         {
 39         	if(data.type == "pictureQueue") {
 40         		returnData = {type : "pictureQueue"};
 41 	            for(var i = 0; i < data.pictureQueue.length; i++) {
 42 	            	returnData[i+1] = getUrlForPicture(data.pictureQueue[i]);
 43 	            }
 44         	} else if(data.type == "finished") {
 45         		returnData = data;
 46         	} else if (data.type == "experimentinstruction") {
 47         		returnData = data;
 48         	}
 49         },
 50         error: function(request, status, error) {
 51             alert(request.responseText);
 52         }
 53     });
 54     return returnData;
 55 }

 57 /**
 58  * Gets URL for a given pictureId together with pictureorder,
 59  * id and url.
 60  * @param pictureId ID of the picture you want data for.
 61  * @return array containing picture information with url, pictureOrder and id.
 62  */
 63 function getUrlForPicture(pictureId) {
 64 	var pictureUrl = new Array();
 65     $.ajax
 66     ({
 67         url: 'functions.php',
 68         async: false,
 69         data: {
 70         	'option':'getPictureUrl',
 71         	'pictureId':pictureId
 72         },
 73         dataType: 'json',
 74         success: function(data)
 75         {
 76             pictureUrl = data;
 77         },
 78         error: function(request, status, error) {
 79             alert(request.responseText);
 80         }
 81     });
 82     return pictureUrl;
 83 }
