128 /**
129  * Checks if the email is valid
130  * @param {string} email the email to check.
131  */
132 function validEmail(email) {
133     var filter = /^\s*[\w\-\+_]+(\.[\w\-\+_]+)*\@[\w\-\+_]+\.[\w\-\+_]+(\.[\w\-\+_]+)*\s*$/;
134     return String(email).search(filter) !== -1;
135 }

225 /**
226  * Checks if assumed old password actually matches the old password.
227  * Checks if new password is longer than four characters.
228  * Checks if the two new password and confirm new password matches.
229  * @returns {Boolean} if all the checks where cleared
230  */
231 function checkValuesPassword() {
232     var sessionPassword = getUserSession('password');               //fetches user password from session
233     var oldPassword = CryptoJS.SHA3($('#password').val()).toString();      //crypts assumed old password from user.
234 
235     var newPassword = $('#password2').val();            //fetching value of new password
236     var confirmNewPassword = $('#password3').val();     //fetching value of confirm new password
237 
238     if (!(sessionPassword == oldPassword)) {     //assumed old password actually matches old passwor
239         $('#password').after('<br><div id="notify"><div class="span3" style="margin: 0 20px"></div>' +
240                 '<div class="bg-red notice marker-on-top span1">' +
241                 'Password does not match current password' +
242                 '</div></div>');
243         return false;
244     }
245 
246     if (newPassword.length <= 4) {                  //checks if length of new password is valid
247         $('#password2').after('<div id="notify"><div class="span3" style="margin: 0 20px"></div>' +
248                 '<div class="bg-red notice marker-on-top span1">' +
249                 'Password needs to be longer than 4 characters' +
250                 '</div></div>');
251         return false;
252     }
253 
254     if (!(newPassword == confirmNewPassword)) {         //if new password and confirm new password matches.
255         $('#password3').after('<div id="notify"><div class="span3" style="margin: 0 20px"></div>' +
256                 '<div class="bg-red notice marker-on-top span1">' +
257                 'Passwords do not match' +
258                 '</div></div>');
259         console.log("Both passwords do not match");
260         return false;
261     }
262 
263     return true;
264 }

289 /**
290  * Checks validity of input from user and if any of the fields have been updated.
291  * @returns {Boolea} whether any fields were to short/invalid
292  */
293 function checkValuesUserInfo() {
294 
295     console.log("Inside UpdateValuesUserInfo");
296 
297     var firstName = $('#first-name').val();
298     var sessionFirstName = getUserSession('firstName');
299 
300     var lastName = $('#last-name').val();
301     var sessionLastName = getUserSession('lastName');
302 
303     var nationality = $('#nationality').val();
304     var sessionNationality = getUserSession('nationality');
305 
306     var title = $('#title').val();
307     var sessionTitle = getUserSession('title');
308 
309     var phoneNumber = $('#phone-number').val();
310     var sessionPhoneNumber = getUserSession('phoneNumber');
311 
312     //checking if there is any new input
313     if (firstName != sessionFirstName) {
314         if (firstName.length <= 1) {
315             $('#first-name').after('<br><div id="notify"><div class="span3" style="margin: 0 20px"></div>' +
316                     '<div class="bg-red notice marker-on-top span1">' +
317                     'Too short' +
318                     '</div></div>');
319             return false;
320         }
321     }
322     //checking if there is any new input and length
323     if (lastName != sessionLastName) {
324         if (lastName.length <= 1) {
325             $('#last-name').after('<br><div id="notify"><div class="span3" style="margin: 0 20px"></div>' +
326                     '<div class="bg-red notice marker-on-top span1">' +
327                     'Too short' +
328                     '</div></div>');
329             return false;
330         }
331     }
332     //checking if there is any new input and length
333     if (nationality != sessionNationality) {
334         if (nationality.length <= 3) {
335             $('#nationality').after('<br><div id="notify"><div class="span3" style="margin: 0 20px"></div>' +
336                     '<div class="bg-red notice marker-on-top span1">' +
337                     'Too short' +
338                     '</div></div>');
339             return false;
340         }
341 
342     }
343     //checking if there is any new input and length
344     if (title != sessionTitle) {
345         if (firstName.length <= 1) {
346             $('#title').after('<br><div id="notify"><div class="span3" style="margin: 0 20px"></div>' +
347                     '<div class="bg-red notice marker-on-top span1">' +
348                     'Too short' +
349                     '</div></div>');
350             return false;
351         }
352     }
353     //checking if there is any new input and length
354     if (phoneNumber != sessionPhoneNumber) {
355         if (firstName.length <= 3) {
356             $('#phone-number').after('<br><div id="notify"><div class="span3" style="margin: 0 20px"></div>' +
357                     '<div class="bg-red notice marker-on-top span1">' +
358                     'Too short' +
359                     '</div></div>');
360             return false;
361         }
362     }
363     //if everything is unchanged
364     if (firstName == sessionFirstName && lastName == sessionLastName && nationality == sessionNationality && title == sessionTitle && phoneNumber == sessionPhoneNumber) {
365         $.Notify({
366             content: "No new info",
367             style: {background: 'red'},
368         });
369         return false;           //do nothing
370     }
371     else {
372         return true;            //all or some information has been updated.
373     }
374 }
