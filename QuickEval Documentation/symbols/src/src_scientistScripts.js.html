1 /**
  2  * Gets imagesets. Sorts them based on parameter, by name if no parameter,
  3  * by date if true
  4  * @param {type} $sort Will sort by date if true
  5  * @returns {data|getImageSets.imageSets}   Two dimensional array containing image sets and data
  6  */
  7 function getImageSets($sort) {
  8     var imageSets;
  9     $.ajax({
 10         url: 'ajax/scientist/getImageSets.php',
 11         async: false,
 12         dataType: 'json',
 13         success: function(data) {
 14             imageSets = data;
 15         },
 16         error: function(request, status, error) {
 17             alert(request.responseText);
 18         }
 19     });
 20     if ($sort == true) { //Returns a unsorted list if sorting by date
 21         return imageSets;
 22     }
 23 
 24     imageSets.sort(function(a, b) { //Sorts the list based on name
 25         if (a.name < b.name) {
 26             return -1;
 27         }
 28         if (a.name > b.name) {
 29             return 1;
 30         }
 31         return 0;
 32     });
 33     return imageSets;
 34 }

 81 /**
 82  * Function makes a new experiment and stores it in the database.
 83  * Stores data from step 1 in experiment creation.
 84  * @return Id of the newly created experiment.
 85  */
 86 function startNewExperiment(name, shortDescription, description, exType,
 87     screenWhitePoint, screenLuminance, roomIllumination, ambientIllumination) {
 88     var experimentId;
 89     $.ajax({
 90         url: 'ajax/scientist/experimentFunctions.php',
 91         async: false,
 92         data: {
 93             'option': 'newExperiment',
 94             'name': name,
 95             'shortDescription': shortDescription,
 96             'description': description,
 97             'exType': exType,
 98             'screenWhitePoint': screenWhitePoint,
 99             'screenLuminance': screenLuminance,
100             'roomIllumination': roomIllumination,
101             'ambientIllumination': ambientIllumination
102         },
103         datatype: 'json',
104         success: function(data) {
105             experimentId = data;
106         },
107         error: function(request, status, error) {
108             alert(request.responseText);
109             experimentId = 0; //If error
110         }
111     });
112     return experimentId.replace(/"/g, "");
113 }
