151 /**
152  * Sets an experiment hidden or public
153  * @param {int} $experimentId Id of experiment
154  * @param {int} $hidden       1 if set to hidden, 0 if set to public
155  */
156 function setExperimentHidden($experimentId, $hidden) {
157     var hiddenType = ($hidden == 0) ? '1 = Public' : '0 = Hidden';
158 
159     $.ajax({
160         type: 'POST',
161         url: 'ajax/scientist/setExperimentHidden.php',
162         data: {
163             'experimentId': $experimentId,
164             'hidden': hiddenType
165         },
166         dataType: 'json',
167         success: function(data) {
168             if (data == 1) { //that particular experiment was successfully deleted.
169                 $.Notify({ //notifies user about successfull experiment deletion change.
170                     content: "Experiment successfully set " + (($hidden == 1) ? "hidden" : "public"),
171                     style: {
172                         background: 'lime'
173                     },
174                 });
175                 setUpExperimentView($experimentId);
176             }
177         },
178         error: function(request, status, error) {
179             alert(request.responseText);
180         }
181     });
182 }

434 function arrayObjectIndexOf(myArray, searchTerm, property) {
435     for (var i = 0, len = myArray.length; i < len; i++) {
436         if (myArray[i][property] === searchTerm)
437             return i;
438     }
439     return -1;
440 }

442 /**
443  * Adds points to table based on parameters
444  * @param  {int} $points      amount of points to add
445  * @param  {int} $firstImage  id of image who won
446  * @param  {int} $secondImage id of image who lost
447  * @param  {JQuery object} $table       jquery object of table where images are
448  * @param  {array} $array       array of image ids
449  */
450 function pairAddPoints($points, $firstImage, $secondImage, $table, $array) {
451     var imageIndex = arrayObjectIndexOf($array, $firstImage, 'id');
452     var wonAgainstIndex = arrayObjectIndexOf($array, $secondImage, 'id');
453 
454     var resultList = $table.find('.result-list');
455     var cell = resultList.find('tr:eq(' + imageIndex + ')').children().eq(wonAgainstIndex + 1);
456     cell.html((cell.html() == "") ? parseFloat($points) : +cell.html() + parseFloat($points));
457 }

460 /**
461  * @author  //http://stackoverflow.com/questions/5223/length-of-javascript-object-ie-associative-array
462  * Returns size of object, how many keys
463  * @param  {object} obj object
464  * @return {int}     size
465  */
466 Object.size = function(obj) {
467     var size = 0,
468         key;
469     for (key in obj) {
470         if (obj.hasOwnProperty(key))
471             size++;
472     }
473     return size;
474 };
