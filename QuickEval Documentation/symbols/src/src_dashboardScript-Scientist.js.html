 52 /**
 53  * Inserts data into Chartjs's function.
 54  * @param {type} dataRank total experiments of Rank order.
 55  * @param {type} hiddenRank how many Rank order experiments are hidden.
 56  * @param {type} dataPair total experiments of Pair comparison.
 57  * @param {type} hiddenPair how many Pair comparison experiments are hidden.
 58  * @param {type} dataCat total experiments of Category.
 59  * @param {type} hiddenCat how Category experiments are hidden.
 60  * @returns {undefined}
 61  */
 62 function setData(dataRank, hiddenRank, dataPair, hiddenPair, dataCat, hiddenCat) {
 63     dataSource = [
 64         {type: "Pair comparison", public: dataPair, hidden: hiddenPair},
 65         {type: "Rank order", public: dataRank, hidden: hiddenRank},
 66         {type: "Category", public: dataCat, hidden: hiddenCat}
 67     ];
 68 }

 70 /**
 71  * Fetches total experiments of all three types and how many of them are hidden.
 72  * @returns {undefined}
 73  */
 74 function getTotalExperiments() {
 75     $.ajax
 76             ({
 77                 url: 'ajax/scientist/getTotalExperiments.php',
 78                 async: false,
 79                 data: {},
 80                 dataType: 'json',
 81                 type: 'post',
 82                 success: function(data) {
 83                     var rating = data.totalRating[0].total;
 84                     var pair = data.totalPair[0].total;
 85                     var category = data.totalCategory[0].total;
 86                     var hiddenRating = data.hiddenRating[0].total;
 87                     var hiddenPair = data.hiddenPair[0].total;
 88                     var hiddenCategory = data.hiddenCategory[0].total;
 89                     var publicRating = rating - hiddenRating;
 90                     var publicPair = pair - hiddenPair;
 91                     var publicCategory = category - hiddenCategory;
 92                     //sends fetched data to function for insertion:
 93                     setData(publicRating, hiddenRating, publicPair, hiddenPair, publicCategory, hiddenCategory);
 94                 },
 95                 error: function(xhr, ajaxOptions, thrownError) {
 96 
 97                 }
 98             });
 99 }

101 /**
102  * Fetches the three latest experiments belonging to user.
103  * Then adds them to a table.
104  * @returns {undefined}
105  */
106 function getExperimentStatistics() {
107     var totalElements;
108     $.ajax
109             ({
110                 url: 'ajax/scientist/getExperimentStatistics.php',
111                 async: false,
112                 data: {},
113                 dataType: 'json',
114                 type: 'post',
115                 success: function(data) {
116                     //checks how many experiments are returned:
117                     if (typeof data.title3 != 'undefined') {
118                         totalElements = 3;
119                     }
120                     else if (typeof data.title2 != 'undefined') {
121                         totalElements = 2;
122                     }
123                     else if (typeof data.title1 != 'undefined') {
124                         totalElements = 1;
125                     }
126                     else {                      //successfully runned backend script, but for some reason returns empty/none experiments
127                         // exits function.
128 
129                         return;
130                     }
131                     //intermediate saving of data for a more clear view of what is happening.
132                     var title1 = data.title1;
133                     var visitors1 = data.visitors1[0].total;
134                     var completion1 = data.completion1[0].total;
135                     var average1 = data.average1;
136                     var title2 = data.title2;
137                     var visitors2 = data.visitors2[0].total;
138                     var completion2 = data.completion2[0].total;
139                     var average2 = data.average2;
140                     var title3 = data.title3;
141                     var visitors3 = data.visitors3[0].total;
142                     var completion3 = data.completion3[0].total;
143                     var average3 = data.average3;
144                     //sets content of variable if they are undefined:
145                     (average1 === undefined) ? average1 = "N/A" : average1;
146                     (average2 === undefined) ? average2 = "N/A" : average2;
147                     (average3 === undefined) ? average3 = "N/A" : average3;
148                     //adding each row to table
149                     $('#table1 tr:last').after('<tr><td>' + title1 + '</td><td>' + visitors1 + '</td><td>' + completion1 + '</td><td>' + average1 + '</td></tr>');
150                     $('#table1 tr:last').after('<tr><td>' + title2 + '</td><td>' + visitors2 + '</td><td>' + completion2 + '</td><td>' + average2 + '</td></tr>');
151                     $('#table1 tr:last').after('<tr><td>' + title3 + '</td><td>' + visitors3 + '</td><td>' + completion3 + '</td><td>' + average3 + '</td></tr>');
152                 },
153                 error: function(xhr, ajaxOptions, thrownError) {
154 //                    console.log("Error");
155 //                    console.log(xhr.status);
156 //                    console.log(thrownError);
157                 }
158             });
159 }
