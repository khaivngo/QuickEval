 62 var runned = 0;
 63 /**
 64  * Goes to next step in current experiment
 65  * @returns {undefined}
 66  */
 67 function loadExperiment2(data) {
 68     var data;
 69 
 70     if (runned < 1) {
 71         data = getNextInExperimentForObserver();
 72     }
 73 
 74     if (data['type'] == "experimentinstruction") {                //is instruction
 75         var instruction = data['experimentinstruction'];
 76         loadInstruction(instruction);
 77         onlyInstruction();
 78         // data = getNextInExperimentForObserver();
 79         nextImageCategory();
 80 
 81     }
 82 
 83     if (data['type'] == "pictureQueue") {                   //picture set 
 84         var originalImageUrl = data[1]['originalUrl'].url; //getting url of original image
 85         loadOriginal(originalImageUrl); // calls function for setting image.
 86 
 87         var reproductionPictureOrder = data[1].pictureOrderId;
 88         var reproductionImageUrl = data[1].url;
 89         
 90         loadCategoryReproduction(reproductionPictureOrder, reproductionImageUrl);
 91     }
 92 
 93     if (data['type'] == "finished") {           //user is finished
 94         experimentComplete(experimentId);
 95         $('#popupButtons3').css({"margin-left": "-3.5%"});
 96         $('#contactArea3').append("You have finished, thank you for your time <br><br> Click Quit to return to front page.");
 97 
 98         centerPopup3();
 99         loadPopup3();
100     }
101 
102     runned = 1;
103 }
