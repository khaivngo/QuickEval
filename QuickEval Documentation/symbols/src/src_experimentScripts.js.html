213 /**
214  * Checks whether user has chosen left or right reproduction and updates DB before loading next
215  * @returns {undefined}
216  */
217 function nextComparison() {
218 
219     var choose;
220     var pictureOrderId;
221 
222     if ($('#left-reproduction').hasClass('main')) {         //user choose left
223         choose = "null";
224         pictureOrderId = $('#left-reproduction-link').attr('pictureOrderId');
225 
226         postResults("pair", pictureOrderId, choose);
227     }
228 
229     if ($('#right-reproduction').hasClass('main')) {        //user choose right
230         choose = "null";
231         pictureOrderId = $('#right-reproduction-link').attr('pictureOrderId');
232 
233         postResults("pair", pictureOrderId, choose);
234     }
235 
236     if (!$('#left-reproduction').hasClass('main') && !$('#right-reproduction').hasClass('main') && runned == 1) {      //user chose none
237         choose = "1";
238         pictureOrderId = $('#left-reproduction-link').attr('pictureorderid');
239         postResults("pair", pictureOrderId, choose);
240     }
241 
242     var data = getNextInExperimentForObserver();
243     nextStep(data);
244 
245     runned = 1;
246 }

313 /**
314  * Receives object with data about experiment of reproductions and instructions.
315  * Dependent of what object contains it either display reproduction, loads instruction 
316  * or if finished sends user back to front page.
317  * @param {type} $imageUrl1 first/left reproduction picture 
318  * @param {type} $imageUrl2 second/right reproducation picture
319  * @returns {undefined}
320  */
321 function nextStep(receivedObject) {
322     var instruction;
323     var data = receivedObject;
324     console.log(receivedObject);
325 
326     if (data['type'] == "experimentinstruction") {     //object contains and instruction
327         instruction = data['experimentinstruction'];
328         setInstruction(instruction);
329         onlyInstruction();
330     }
331 
332     if (data['type'] == "pictureQueue") {               //received array contains data about the reproduction
333         var originalUrl = data[1]['originalUrl'].url; //getting url of original image
334         setOriginal(originalUrl); // calls function for setting image.
335 
336         setPictures(data);
337         setOriginal();
338     }
339 
340     if (data['type'] == "finished") {           //user is finished
341         experimentComplete(experimentId);
342         $('#contactArea').empty();
343 
344         $('#button-next').text('Quit');
345         activateNextButton();
346         $('#button-next').show();
347 
348         $('#contactArea').append("You have finished, thank you for your time <br><br> Click Quit to return to front page.");
349         $('#continue').hide();
350         $('#popupButtons').css("margin-left", "46%");
351         $('#cancel-experiment').trigger('click');
352     }
353 }
