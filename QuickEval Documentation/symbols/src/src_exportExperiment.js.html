  5 /**
  6  * Will start the exporting process for an experiment, and force a download to the user.
  7  * @param experimentId is the ID for the experiment which to export.
  8  * @return link = string with downloadPath.
  9  */
 10 function generateSQLForExperiment(experimentId) {
 11 	
 12 	var link;
 13     $.ajax
 14     ({
 15         url: 'ajax/scientist/exportExperiment.php',
 16         async: false,
 17         data: {
 18         	'option':'generateSQL',
 19         	'experimentId':experimentId,
 20         },
 21         dataType: 'json',
 22         success: function(data)
 23         {
 24         	link = "<a href=ajax/scientist/" + data + ">Download Experiment</a>";
 25 			$('#wrapper').append("<div style=display:none;>" + 
 26 		    "<iframe id=frmDld src= ajax/scientist/" + data + "></iframe></div>");
 27         },
 28         error: function(request, status, error) {
 29             alert(request.responseText);
 30         }
 31     });
 32     return link;
 33 }
 34 /**
 35  * Will generate and show the uploader for importing experiments.
 36  */
 37 function generateUploaderImportExperiment() {
 38 	$('#right-panel').empty();
 39 	$('#right-panel').append("<p class ='item-title-secondary'>Upload your earlier exported experiment.\n</p>");
 40 	$('#right-panel').append("<div id=filelist>Your browser doesn't have Flash, Silverlight or HTML5 support.</div>" +
 41 			"<br />" +
 42 			"<div id=container>" +
 43 			    "<button class ='primary' id=pickfiles href=javascript:;>Select file</button>    " +
 44 			    "<button class ='primary' id=uploadfiles href=javascript:;>Upload file</button>"  +
 45 			"</div>" + 
 46 			"<br />" +
 47 			"<pre id=console></pre>");
 48 	
 49 	var uploader = new plupload.Uploader({
 50 	    runtimes : 'html5,flash,silverlight,html4',
 51 	     
 52 	    browse_button : 'pickfiles', // you can pass in id...
 53 	    container: document.getElementById('container'), // ... or DOM Element itself
 54 	    chunk_size: '200kb',
 55 
 56 	     
 57 	    url : "ajax/scientist/uploadExperiment.php",
 58 	     
 59 	    filters : {
 60 	        max_file_size : '2048mb',
 61 	        mime_types: [
 62 	            {title : "Zip files", extensions : "zip"}
 63 	        ],
 64 	        multi_selection:false
 65 	        	
 66 	    },
 67 	 
 68 	    // Flash settings
 69 	    flash_swf_url : '/plupload/js/Moxie.swf',
 70 	 
 71 	    // Silverlight settings
 72 	    silverlight_xap_url : '/plupload/js/Moxie.xap',
 73 	     
 74 	 
 75 	    init: {
 76 	        PostInit: function() {
 77 	            document.getElementById('filelist').innerHTML = '';
 78 	 
 79 	            document.getElementById('uploadfiles').onclick = function() {
 80 	                uploader.start();
 81 	                return false;
 82 	            };
 83 	        },
 84 	 
 85 	        FilesAdded: function(up, files) {
 86 	        	var maxFiles = 1;
 87 	        	console.log("[up]", up);
 88 	        	console.log("[files]", files);	
 89 	    	    if(up.files.length > maxFiles)
 90 	    	    {
 91 	    	    	alert("Only 1 file!");
 92 	    	    	generateUploaderImportExperiment();
 93 	    	    	
 94 	    	    }
 95 	            plupload.each(files, function(file) {
 96 	                document.getElementById('filelist').innerHTML += '<div id="' + file.id + '">' + file.name + ' (' + plupload.formatSize(file.size) + ') <b></b></div>';
 97 	            });
 98 	        },
 99 	        
100             UploadComplete: function(up, files) {
101                 // Called when all files are either uploaded or failed
102                 //console.log('[UploadComplete]');
103             	importExperiment();
104             },
105             
106 	 
107 	        UploadProgress: function(up, file) {
108 	            document.getElementById(file.id).getElementsByTagName('b')[0].innerHTML = '<span>' + file.percent + "%</span>";
109 	        },
110 	 
111 	        Error: function(up, err) {
112 	            document.getElementById('console').innerHTML += "\nError #" + err.code + ": " + err.message;
113 	        }
114 	    }
115 	});
116 
117 	uploader.init(); 
118 }
119 /**
120  * Imports the experiment.  This function is to run AFTER the zip file has been uploaded.
121  */
122 function importExperiment() {
123     $.ajax
124     ({
125         url: 'ajax/scientist/importExperiment.php',
126         async: false,
127         data: {
128         	'option':'generateExperiment',
129         },
130         dataType: 'json',
131         success: function(data)
132         {
133         	if(data == 1) {
134         	generateUploaderImportExperiment();
135         	$('#right-panel').append("" +
136         			"<div class='notice marker-on-top' style='width:250px';>" +
137         			"<div class='fg-white'>Experiment successfully imported!</br>Import another?</div></div>");
138         	} else {
139             	$('#right-panel').append("" +
140             			"<div class='notice marker-on-top bg-darkRed' style='width:250px';>" +
141             			"<div class='fg-white'>Something went wrong,are you sure it</br> is the correct file?</div></div>");
142         	}
143         },
144         error: function(request, status, error) {
145         	cleanUpDirectory();
146           //  alert(request.responseText);
147         	$('#right-panel').empty();
148         	generateUploaderImportExperiment();
149         	$('#right-panel').append("" +
150         			"<div id='error-message' class='notice marker-on-top bg-darkRed' style='width:250px';>" +
151         			"<div class='fg-white'>Something went wrong,are you sure it is the correct file?</div></div>");
152         	$('#error-message').fadeIn("fast", function(){        
153                 $("#error-message").fadeOut(7000);
154             });
155 
156         }
157     });
158 }
