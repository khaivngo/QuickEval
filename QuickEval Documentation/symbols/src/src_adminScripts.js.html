 85 /**
 86  * Checks if the string is an valid email address
 87  * @param {type} email the string to be matched
 88  * @returns {Boolean}
 89  */
 90 function validEmail(email) {
 91     var filter = /^\s*[\w\-\+_]+(\.[\w\-\+_]+)*\@[\w\-\+_]+\.[\w\-\+_]+(\.[\w\-\+_]+)*\s*$/;
 92     return String(email).search(filter) !== -1;
 93 }

372 /**
373  * Fills the input field with matching emails from database
374  * @returns {undefined}
375  */
376 function loadUsers() {
377     $.ajax({
378         type: 'POST',
379         url: 'ajax/admin/getUsers.php',
380         data: {},
381         dataType: 'json',
382         success: function(data) {
383             var users = [];
384             for (var i = 0; i < data.length; i++) {         //iterates all returned users
385                 var buffer = data[i]['email'];              //saves next email to a temporary variable,
386                 users.push(buffer);                         // then pushes it on the array.
387             }
388             $("#change-access-user").autocomplete({
389                 source: users
390             });
391         },
392         error: function(request, status, error) {
393             alert(request.responseText);
394             console.log(request.responseText);
395         }
396     });
